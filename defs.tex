\tikzset{
	visible on/.style={alt={#1{}{transparent}}},
	dim/.style={alt={#1{opacity = 0.3}{}}},
	alt/.code args={<#1>#2#3}{%
		\alt<#1>{\pgfkeysalso{#2}}{\pgfkeysalso{#3}} % \pgfkeysalso doesn't change the path
	},
}

	},
